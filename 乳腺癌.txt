# 导入数据处理、可视化、机器学习相关库，并解决图表中文乱码问题。

# 数据处理库
import kagglehub
import pandas as pd
import numpy as np

# 可视化库
import matplotlib.pyplot as plt
import seaborn as sns

# 模型与评估工具
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
from sklearn.metrics import accuracy_score, classification_report, confusion_matrix

# 分类模型
from sklearn.linear_model import LogisticRegression
from sklearn.svm import SVC
from sklearn.tree import DecisionTreeClassifier
from sklearn.ensemble import RandomForestClassifier

# 解决 Notebook 中文乱码
plt.rcParams["font.sans-serif"] = ["SimHei"]
plt.rcParams["axes.unicode_minus"] = False

# 自动从 Kaggle 下载乳腺癌数据集，查看数据规模、样本内容、数据类型与统计分布，初步了解数据特征

# 下载 Kaggle 乳腺癌数据集
path = kagglehub.dataset_download("uciml/breast-cancer-wisconsin-data")
print("数据集存放路径：", path)

# 读取数据
df = pd.read_csv(f"{path}/data.csv")

# 查看基础信息
print("=== 数据集形状 ===")
print(df.shape)
print("\n=== 前5行数据 ===")
display(df.head())
print("\n=== 数据类型与缺失值 ===")
df.info()
print("\n=== 数据统计描述 ===")
display(df.describe())

# 数据清洗、标签编码、划分训练集 / 测试集，并对特征做标准化处理（适配线性类模型）。
# 剔除无意义字段：id、空值列
# 标签转换：恶性 (M)→1，良性 (B)→0
# 按照 7:3 比例划分训练集与测试集
# 特征标准化，消除量纲影响

# 1. 删除无用列
df.drop(["id", "Unnamed: 32"], axis=1, inplace=True)

# 2. 标签映射：良性B=0，恶性M=1
df["diagnosis"] = df["diagnosis"].map({"M": 1, "B": 0})

# 3. 划分特征和标签
X = df.drop("diagnosis", axis=1)
y = df["diagnosis"]

# 4. 划分训练集、测试集 7:3
X_train, X_test, y_train, y_test = train_test_split(
    X, y, test_size=0.3, random_state=42, stratify=y
)

# 5. 特征标准化（供逻辑回归、SVM 使用）
scaler = StandardScaler()
X_train_scaled = scaler.fit_transform(X_train)
X_test_scaled = scaler.transform(X_test)

print("训练集特征维度：", X_train.shape)
print("测试集特征维度：", X_test.shape)

# 选用逻辑回归、SVM、决策树、随机森林四种经典分类算法进行训练。分别计算模型准确率、精确率、召回率等指标，并对模型效果进行排序对比。

# 定义模型集合
models = {
    "逻辑回归": LogisticRegression(random_state=42),
    "支持向量机SVM": SVC(random_state=42),
    "决策树": DecisionTreeClassifier(random_state=42),
    "随机森林": RandomForestClassifier(random_state=42, n_estimators=100)
}

# 保存各模型准确率
model_results = {}

print("===== 各模型分类评估结果 =====\n")
for name, model in models.items():
    # 区分是否使用标准化数据
    if name in ["逻辑回归", "支持向量机SVM"]:
        model.fit(X_train_scaled, y_train)
        y_pred = model.predict(X_test_scaled)
    else:
        model.fit(X_train, y_train)
        y_pred = model.predict(X_test)
    
    # 计算准确率
    acc = accuracy_score(y_test, y_pred)
    model_results[name] = acc
    
    # 输出评估报告
    print(f"【{name}】 准确率: {acc:.4f}")
    print(classification_report(y_test, y_pred, target_names=["良性(B)", "恶性(M)"]))
    print("-" * 60)

# 按准确率排序输出排名
print("===== 模型准确率排名（从高到低） =====")
sorted_res = sorted(model_results.items(), key=lambda x: x[1], reverse=True)
for idx, (name, acc) in enumerate(sorted_res, 1):
    print(f"{idx}. {name}  →  准确率：{acc:.4f}")

# 柱状图：直观对比所有模型的准确率高低
# 混淆矩阵：展示最优模型在良性 / 恶性样本上的预测正误情况

# 1. 不同模型准确率柱状图
plt.figure(figsize=(10, 5))
sns.barplot(x=list(model_results.keys()), y=list(model_results.values()))
plt.title("不同分类模型准确率对比", fontsize=14)
plt.ylabel("准确率", fontsize=12)
plt.ylim(0.8, 1.0)
plt.tight_layout()
plt.show()

# 2. 最优模型（随机森林）混淆矩阵
best_model = models["随机森林"]
y_best_pred = best_model.predict(X_test)
cm = confusion_matrix(y_test, y_best_pred)

plt.figure(figsize=(6, 5))
sns.heatmap(cm, annot=True, fmt="d", cmap="Blues",
            xticklabels=["良性(B)", "恶性(M)"],
            yticklabels=["良性(B)", "恶性(M)"])
plt.title("随机森林 混淆矩阵", fontsize=14)
plt.ylabel("真实标签")
plt.xlabel("预测标签")
plt.tight_layout()
plt.show()

# 分析各特征之间的相关程度，了解数据内在关联，辅助理解数据分布特点。

# 特征相关性热力图
plt.figure(figsize=(14, 12))
corr = df.corr()
sns.heatmap(corr, cmap="coolwarm", annot=False)
plt.title("特征相关性热力图", fontsize=14)
plt.tight_layout()
plt.show()
